  <action>
    Create `tests/integration/platformio-cli.test.ts`. Use Node's `child_process.execSync` (wrapped in try/catch) to run real CLI commands. All three tests must pass without hardware.

    ```typescript
    import { execSync } from 'child_process';
    import { mkdtempSync, rmSync, writeFileSync } from 'fs';
    import { tmpdir } from 'os';
    import { join } from 'path';
    import { afterAll, beforeAll, describe, expect, it } from 'vitest';

    const PIO_TIMEOUT = 120_000; // 2 min — board index download can be slow

    describe('PlatformIO CLI integration', () => {
      it('pio --version returns a version string', () => {
        const out = execSync('pio --version', { encoding: 'utf8', timeout: 10_000 });
        expect(out).toMatch(/PlatformIO Core/i);
      });

      it('pio boards returns at least one board entry', () => {
        const out = execSync('pio boards --json-output', {
          encoding: 'utf8',
          timeout: PIO_TIMEOUT,
        });
        const boards = JSON.parse(out);
        expect(Array.isArray(boards)).toBe(true);
        expect(boards.length).toBeGreaterThan(0);
      });

      describe('temp project init + build', () => {
        let tmpDir: string;

        beforeAll(() => {
          tmpDir = mkdtempSync(join(tmpdir(), 'pio-mcp-test-'));
        });

        afterAll(() => {
          rmSync(tmpDir, { recursive: true, force: true });
        });

        it('pio init and pio run succeed on a minimal esp32dev project', () => {
          // Init project
          execSync('pio init --board esp32dev --project-option "framework=arduino"', {
            cwd: tmpDir,
            encoding: 'utf8',
            timeout: PIO_TIMEOUT,
          });

          // Write minimal firmware so build has something to compile
          writeFileSync(
            join(tmpDir, 'src', 'main.cpp'),
            '#include <Arduino.h>\nvoid setup(){}\nvoid loop(){}\n'
          );

          // Build — no device needed
          const buildOut = execSync('pio run', {
            cwd: tmpDir,
            encoding: 'utf8',
            timeout: PIO_TIMEOUT,
          });

          expect(buildOut).toMatch(/SUCCESS/i);
        });
      });
    });
    ```

    The test file lives under `tests/integration/` (create the directory). No mock — these are real CLI calls.
